  <script>
  async function loadFeedback() {
    const container = document.getElementById('feedbackContainer');
    try {
      const res = await fetch('/.netlify/functions/feedback');
      const data = await res.json();
      container.innerHTML = '';

      if (!Array.isArray(data) || data.length === 0) {
        container.innerHTML = '<p style="text-align:center;color:#aaa;">Noch kein Feedback vorhanden üí≠</p>';
        return;
      }

      data.reverse().forEach(item => {
        const card = document.createElement('div');
        card.className = 'feedback-card';
        card.innerHTML = `
          <div class="feedback-name">${item.name}</div>
          <div class="feedback-msg">${item.message}</div>
          <div class="feedback-date">${new Date(item.timestamp).toLocaleString('de-DE')}</div>
        `;
        container.appendChild(card);
      });
    } catch (err) {
      console.error(err);
      container.innerHTML = '<p style="color:#ff6b6b;text-align:center;">Fehler beim Laden des Feedbacks ‚ùå</p>';
    }
  }
  loadFeedback();
  </script>
